(module processador-neural racket
    (provide treinar-neuronio processar-camada)

    (define (funcao-ativacao x)
        (if (> x 0)
            x
            0
        )
    )

    (define (treinar-neuronio peso entrada bias)
        (begin
            (set! peso (+ peso (* entrada 0.01)))
            (funcao-ativacao (+ (* peso entrada) bias))
        )
    )

    (define (processar-camada lista-pesos entrada bias)
        (if (>= (calcula-comprimento lista-pesos) 1)
            (list 
                (treinar-neuronio 0.5 entrada bias)
                "processamento-em-curso"
            )
            "camada-vazia"
        )
    )
)

(define (loop-principal iteracoes)
    (if (> iteracoes 0)
        (begin
            (display "Iterando...")
            (loop-principal (- iteracoes 1))
        )
        "fim-do-treino"
    )
)

(define resultado (loop-principal 100))
